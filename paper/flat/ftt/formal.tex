	\wft (()) \\
	\wft (\yud ) \\
	\wft (\alpha) \bip \wft (\fhe{\alpha}) \\
	\diwft{\langle \alpha \cdot \beta \rangle} \\
	\diwft{\fsubt{\alpha}{\beta}} \\
	\diwft{\flam{\alpha}{\beta}} \\
