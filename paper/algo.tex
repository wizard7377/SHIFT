\newcommand{\llparseA}{
	\begin{algorithmic}
		\Procedure{parse}{\pin}
			\State $\prs \gets \emptylist$
			\Repeat 
				\State $(\pin,\prs) \gets E(\pin,\prs)$
			\Until{$\pin \equiv \emptylist$}
			
			\Return \prs
		\EndProcedure
			\State
		\Function{E}{\pin,\prs}
			\If{$\pin [0] \in "0".."9"$}
				\State \Return $E_{\#} (\pin,\prs)$
			\ElsIf{$\pin [0] \equiv \texttt{"+"}$} 
				\State \Return $E_{+} (\pin,\prs)$
			\ElsIf{$\pin [0] \equiv \texttt{"*"}$}
				\State \Return $E_{*} (\pin,\prs)$
			\Else 
				\State \Return error
			\EndIf
		\EndFunction
		\Function{$E_+$}{\pin,\prs}
			\State $\pop (\pin)$
			\State $(\pin,\prs) \gets E(\pin,\prs)$
			\State \Return $E(\pin,\prs)$
		\EndFunction
		\Function{$E_*$}{\pin,\prs}
			\State $\pop (\pin)$
			\State $(\pin,\prs) \gets E(\pin,\prs)$
			\State \Return $E(\pin,\prs)$
		\EndFunction
		\Function{$E_{\#}$}{\pin,\prs}
			\State $a \gets 0$
			\While{\pin [0] \in "0".."9"}
				\State $a \gets a * 10$
				\State $v \gets \mathbb{Z}(\pop(\pin))$
				\State $a \gets a + v$
			\EndWhile
			\State \pushto{a}{\prs}
			\State \Return (\pin,\prs)
		\EndFunction
	\end{algorithmic}
}
